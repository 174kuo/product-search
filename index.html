function doGet() { const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheets()[0]; const data = sheet.getDataRange().getValues(); const rows = data.slice(1); // 跳過標題列 const result = {}; rows.forEach(row => { const id = String(row[0] || '').trim(); if (!id) return; // 日期欄如果是 Date 物件，手動格式化 let date = ''; if (row[1] instanceof Date) { const d = row[1]; const y = d.getFullYear(); const m = String(d.getMonth() + 1).padStart(2, '0'); const day = String(d.getDate()).padStart(2, '0'); date = `${y}/${m}/${day}`; } else { date = String(row[1] || '').trim(); } const batch = String(row[2] || '').trim(); const qty = Number(row[3]) || 0; const cost = Number(row[4]) || 0; const note = String(row[5] || '').trim(); if (!result[id]) result[id] = []; result[id].push({ date, batch, qty, cost, note }); }); return ContentService .createTextOutput(JSON.stringify(result)) .setMimeType(ContentService.MimeType.JSON); }
